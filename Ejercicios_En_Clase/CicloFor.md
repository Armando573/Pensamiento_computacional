Algoritmo Ciclo_for
	nombre <- char
	numveces <- int
	i <- int
	
	Escribir "Ingresa tu nombre"
	Leer nombre
	escribir "Cuentas veces quieres escribir tu nombre?"
	Leer numveces
	
Para i <- 0 Hasta numveces	Con Paso 1 hacer
		escribir nombre
	FinPara
	
FinAlgoritmo
