Algoritmo Ciclo_While
	nombre <- char
	i <- int
	parar <- "n"
	
	Escribir "Ingresa tu nombre"
	Leer nombre
	
	Escribir "Quieres imprimir tu nombre? S_ N_" 
	Leer parar

	Mientras parar <> "n" Hacer
		
		Escribir nombre
		Escribir "Quieres imprimir tu nombre? S_ N_" 
		Leer parar
		
	Fin Mientras
	
	
FinAlgoritmo
